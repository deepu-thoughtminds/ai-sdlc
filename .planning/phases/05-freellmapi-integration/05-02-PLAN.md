   def test_smoke_freellmapi_real_response():
       """Smoke test: send a real /v1/chat/completions request; verify non-stub response.

       This test is skipped in CI unless FREELLMAPI_API_KEY is explicitly set.
       To run locally: FREELLMAPI_API_KEY=<key> pytest tests/test_smoke_freellmapi.py -v -s
       """
       payload = {
           "model": "auto",
           "messages": [{"role": "user", "content": "Reply with exactly the word PONG and nothing else."}],
       }
       try:
           resp = httpx.post(
               f"{FREELLMAPI_BASE_URL}/v1/chat/completions",
               headers={"Authorization": f"Bearer {FREELLMAPI_API_KEY}"},
               json=payload,
               timeout=60.0,
