async def _run_scan_background(project_id: int) -> None:
    """Run codebase scan in background after project save — SCAN-01.

    Opens a fresh AsyncSession-compatible sync session, decrypts credentials
    at call time, and fires scan_and_commit_codebase(). Errors are logged
    but do not propagate — scan failure must not break the create/update response.

    T-18-01: github_token decrypted here and passed immediately; never stored
             in intermediate variable beyond this scope.
    """
    from database import SessionLocal
    try:
        db = SessionLocal()
        project = db.get(Project, project_id)
        if project is None:
            logger.warning("scan background: project %d not found", project_id)
            return
        github_repo_plain = decrypt_credential(project.github_repo)
        github_token_plain = decrypt_credential(project.github_token)
        if not github_repo_plain:
            return
        logger.info("starting codebase scan for project %d (%s)", project_id, github_repo_plain)
        sha = await asyncio.get_event_loop().run_in_executor(
            None,
            scan_and_commit_codebase,
            github_repo_plain,
            github_token_plain,
        )
        logger.info("codebase scan complete for project %d sha=%s", project_id, sha)
    except Exception as exc:
        logger.warning("codebase scan failed for project %d: %s", project_id, exc)
    finally:
        try:
            db.close()
        except Exception:
            pass
```

Note: scan_and_commit_codebase is synchronous (subprocess-based). Run it in a thread executor so it does not block the event loop. Use asyncio.get_event_loop().run_in_executor(None, ...) with positional args.

T-18-04: github_token_plain is a local variable that goes out of scope when _run_scan_background returns; never stored on a module or class attribute.

@router.put("/projects/{project_id}", response_model=ProjectPublic)
def update_project(
    project_id: int,
    payload: ProjectCreate,
    db: Session = Depends(get_db),
) -> ProjectPublic:
    """Update an existing project, re-encrypting all credential and github_repo fields.

    Returns ProjectPublic with HTTP 200. Returns HTTP 404 if project not found.
    Triggers codebase scan if github_repo is set in the updated payload (SCAN-01).

    T-02-02: No token fields returned in response.
    T-02-04: Only project id logged — no token values.
    T-15-02: github_repo stored as ciphertext; returned decrypted via _project_to_public.
    """
    project = db.get(Project, project_id)
    if project is None:
        raise HTTPException(status_code=404, detail=f"Project {project_id} not found")

    project.name = payload.name
    project.project_key = payload.project_key
    project.jira_url = str(payload.jira_url)
    project.jira_email = payload.jira_email
    project.confluence_url = str(payload.confluence_url)
    project.jira_token = encrypt_credential(payload.jira_token)
    project.github_token = encrypt_credential(payload.github_token)
    project.confluence_token = encrypt_credential(payload.confluence_token)
    project.github_repo = encrypt_credential(payload.github_repo)

    db.commit()
    db.refresh(project)
    logger.info("project updated id=%d", project.id)  # T-02-04: no token values logged

    if payload.github_repo:
        asyncio.create_task(_run_scan_background(project.id))

    return _project_to_public(project)
```
