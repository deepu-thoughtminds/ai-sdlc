           try:
               graph_result = await asyncio.to_thread(
                   cbm_call,
                   "search_graph",
                   {"query": issue_summary, "limit": 20},

                   )
               else:
                   graph_text = "(no graph context available)"
           except Exception as exc:
               logger.warning("cbm search_graph failed for %s: %s", issue_key, exc)
           return graph_text
